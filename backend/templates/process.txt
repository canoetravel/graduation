#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>

int main() {
    pid_t pid = fork();
    
    if (pid < 0) {
        // fork失败
        printf("Fork failed!\n");
        return 1;
    } else if (pid == 0) {
        // 子进程：计算斐波那契数列
        printf("Child Process: Calculating Fibonacci...\n");
        
        int fib[10];
        fib[0] = 0;
        fib[1] = 1;
        
        for (int i = 2; i < 10; i++) {
            fib[i] = fib[i-1] + fib[i-2];
        }
        
        printf("Fibonacci sequence (first 10): ");
        for (int i = 0; i < 10; i++) {
            printf("%d ", fib[i]);
        }
        printf("\n");
        
    } else {
        // 父进程：计算阶乘
        printf("Parent Process: Calculating Factorial...\n");
        
        int fact = 1;
        for (int i = 1; i <= 5; i++) {
            fact *= i;
        }
        printf("5! = %d\n", fact);
        
        wait(NULL);
        printf("Child process finished.\n");
    }
    
    return 0;
}
